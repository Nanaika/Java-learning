public class Parser.Parser {

    private int NumbOfWords(int numbOfWords , String names) {
        for (int i = 0; i < names.length(); i++) {
            if (names.charAt(i) == ',') {
                numbOfWords++;
            }
        }
        return numbOfWords;
    }

    public String[] createArrayOfNames(String names) {
        int count = 0;
        int index = 0;
        String name = "";
        int numbOfWords = 0;

        numbOfWords = NumbOfWords(numbOfWords , names);

        String[] arrayOfNames = new String[numbOfWords + 1];

        for (int x = count; x < names.length(); x++) {
            if (names.charAt(x) != ',') {
                name += names.charAt(x);
            } else {
                arrayOfNames[index] = name;
                index++;
                count = x + 1;
                name = "";
            }
        }
        arrayOfNames[index] = name;
        return arrayOfNames;

    }

    public void PrintArrayOfNames(String names) {
        String[] array = createArrayOfNames(names);
        for (String s : array) {
            System.out.println(s);
        }
    }

}
